
service Confluence {
    url: "http://confluence-burrito.ngrok.io/graphql"

    type Page {
       id: ID!
       authorName: String => from user(userName) with input authorName as author
       title: String
    }

    type User {
       id: ID!
       userName: String => from authoredPages(userName) with input userName as authoredPages added
#       _=> from reportedIssues(userName) with input userName as reportedIssues
    }

    type Query {
        page(id:ID!): Page
        user(id: ID, userName: String): User
        authoredPages(userName: ID!): [Page]
    }
}

service Jira {
    url: "https://spring-graphql-proxy.ap-southeast-2.dev.atl-paas.net/graphql"

    type Issue {
       id: ID!
    }

    type Query {
      issue(id: ID!): Issue
      reportedIssues(username: ID!): [Issue]
    }
}

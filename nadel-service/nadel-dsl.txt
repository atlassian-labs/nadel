
service Confluence {
    url: "http://confluence-burrito.ngrok.io/graphql"

    type Page {
       id: ID!
       authorId: String
       title: String
    }

    type User {
       id: ID!
       username: String
       authoredPages: [Page]
       _=> from reportedIssues with username as reportedIssues
    }

    type Query {
        page(id:ID!): Page
        user(id: ID, userName: String): User
        authoredPages(userId: ID!): [Page]
    }
}

service Jira {
    url: "https://spring-graphql-proxy.ap-southeast-2.dev.atl-paas.net/graphql"

    type Issue {
       id: ID!
    }

    type Query {
      issue(id: ID!): Issue
      reportedIssues(username: ID!): [Issue]
    }
}
